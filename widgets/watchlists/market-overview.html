Market Overview Widget

Обзор рынка по категориям с индексами и криптовалютами.

Код для встраивания:

<div id="widget-container"></div>
<script type="text/javascript" src="https://s3.tradingview.com/tv.js"></script>
<script type="text/javascript">
new TradingView.widget({
    "width": "100%",
    "height": 600,
    "locale": "en",
    "darkMode": true,
    "groups": [
        {
            "title": "Indices",
            "symbols": [
                ["S&P 500", "OANDA:SPX500USD"],
                ["NASDAQ", "OANDA:NS100USD"],
                ["Dow Jones", "OANDA:DJIUSD"]
            ],
            "showSymbolLogo": true,
            "showSymbolLabel": true
        },
        {
            "title": "Crypto",
            "symbols": [
                ["Bitcoin", "BITSTAMP:BTCUSD"],
                ["Ethereum", "BITSTAMP:ETHUSD"]
            ],
            "showSymbolLogo": true,
            "showSymbolLabel": true
        }
    ]
});
</script>
